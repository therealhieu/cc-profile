**Goal:** Add testable services for profile, env var, Claude args, and launch behavior.

**Architecture:** Services mutate `Config` directly and keep validation outside the UI. The launch service builds a pure `CommandSpec` for tests, then converts it to `std::process::Command` only at the execution boundary.

**Tech Stack:** Rust 2021, `anyhow`, `serde`, `BTreeMap`, `std::process::Command`, `cargo-nextest`.

use crate::config::validation::{validate_profile_name, validate_required_value};
use crate::config::{Config, Profile};
use anyhow::{Result, bail};

pub fn create_profile(config: &mut Config, name: &str, profile: Profile, set_active: bool) -> Result<()> {
    validate_profile_name(name)?;
    validate_profile(&profile)?;

    if config.profiles.contains_key(name) {
        bail!("Profile '{name}' already exists");
    }

    config.profiles.insert(name.to_string(), profile);
    if set_active {
        config.active_profile = Some(name.to_string());
    }
    Ok(())
}

pub fn set_active_profile(config: &mut Config, name: &str) -> Result<()> {
    if !config.profiles.contains_key(name) {
        bail!("Profile '{name}' does not exist");
    }
    config.active_profile = Some(name.to_string());
    Ok(())
}

pub fn rename_profile(config: &mut Config, old_name: &str, new_name: &str) -> Result<()> {
    validate_profile_name(new_name)?;
    if config.profiles.contains_key(new_name) {
        bail!("Profile '{new_name}' already exists");
    }
    let profile = config
        .profiles
        .remove(old_name)
        .ok_or_else(|| anyhow::anyhow!("Profile '{old_name}' does not exist"))?;
    config.profiles.insert(new_name.to_string(), profile);
    if config.active_profile.as_deref() == Some(old_name) {
        config.active_profile = Some(new_name.to_string());
    }
    Ok(())
}

pub fn update_profile(config: &mut Config, name: &str, profile: Profile) -> Result<()> {
    validate_profile(&profile)?;
    let slot = config
        .profiles
        .get_mut(name)
        .ok_or_else(|| anyhow::anyhow!("Profile '{name}' does not exist"))?;
    *slot = profile;
    Ok(())
}

pub fn delete_profile(config: &mut Config, name: &str) -> Result<()> {
    config
        .profiles
        .remove(name)
        .ok_or_else(|| anyhow::anyhow!("Profile '{name}' does not exist"))?;
    if config.active_profile.as_deref() == Some(name) {
        config.active_profile = None;
    }
    Ok(())
}

fn validate_profile(profile: &Profile) -> Result<()> {
    validate_required_value("Endpoint", &profile.endpoint)?;
    validate_required_value("API key", &profile.api_key)?;
    validate_required_value("Fable model", &profile.fable)?;
    validate_required_value("Opus model", &profile.opus)?;
    validate_required_value("Sonnet model", &profile.sonnet)?;
    validate_required_value("Haiku model", &profile.haiku)?;
    Ok(())
}
```

use crate::config::Config;
use anyhow::{Context, Result, bail};
use std::collections::BTreeMap;
use std::process::Command;

#[derive(Debug, Clone, PartialEq, Eq)]
pub struct CommandSpec {
    pub program: String,
    pub args: Vec<String>,
    pub envs: BTreeMap<String, String>,
}

pub fn build_command_spec(config: &Config) -> Result<CommandSpec> {
    let active_name = config
        .active_profile
        .as_ref()
        .ok_or_else(|| anyhow::anyhow!("No active profile is set"))?;
    let profile = config
        .profiles
        .get(active_name)
        .ok_or_else(|| anyhow::anyhow!("Active profile '{active_name}' does not exist"))?;

    let mut envs = config.envs.clone();
    envs.insert("ANTHROPIC_BASE_URL".to_string(), profile.endpoint.clone());
    envs.insert("ANTHROPIC_API_KEY".to_string(), profile.api_key.clone());
    envs.insert("ANTHROPIC_DEFAULT_FABLE_MODEL".to_string(), profile.fable.clone());
    envs.insert("ANTHROPIC_DEFAULT_OPUS_MODEL".to_string(), profile.opus.clone());
    envs.insert("ANTHROPIC_DEFAULT_SONNET_MODEL".to_string(), profile.sonnet.clone());
    envs.insert("ANTHROPIC_DEFAULT_HAIKU_MODEL".to_string(), profile.haiku.clone());

    let mut args = Vec::new();
    if config.args.dangerously_skip_permissions {
        args.push("--dangerously-skip-permissions".to_string());
    }

    Ok(CommandSpec {
        program: "claude".to_string(),
        args,
        envs,
    })
}

pub fn run_command_spec(spec: &CommandSpec) -> Result<()> {
    let status = Command::new(&spec.program)
        .args(&spec.args)
        .envs(&spec.envs)
        .status()
        .with_context(|| format!("Could not find `{}` on PATH. Please install Claude Code or ensure the `{}` command is available.", spec.program, spec.program))?;

    if !status.success() {
        bail!("{} exited with status: {}", spec.program, status);
    }

    Ok(())
}

pub fn start_claude(config: &Config) -> Result<()> {
    let spec = build_command_spec(config)?;
    run_command_spec(&spec)
}
```
